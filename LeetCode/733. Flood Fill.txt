function floodFill(image, sr, sc, newColor) {
    const oldColor = image[sr][sc];
    
    if (oldColor === newColor) return image;

    const fill = (r, c) => {
        // Check boundaries and if the current pixel has the same color
        if (r < 0 || r >= image.length || c < 0 || c >= image[0].length || image[r][c] !== oldColor) {
            return;
        }
        
        // Change the current pixel to the new color
        image[r][c] = newColor;

        // Perform flood fill in all 4 directions (up, down, left, right)
        fill(r + 1, c);
        fill(r - 1, c);
        fill(r, c + 1);
        fill(r, c - 1);
    };

    // Start flood fill from the given starting pixel
    fill(sr, sc);

    return image;
}
