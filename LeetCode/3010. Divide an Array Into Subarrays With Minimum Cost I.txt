function minCost(nums) {
    let n = nums.length;
    let minCost = Infinity;

    for (let i = 0; i < n - 2; i++) {
        for (let j = i + 1; j < n - 1; j++) {
            let cost = nums[0] + nums[i + 1] + nums[j + 1];
            minCost = Math.min(minCost, cost);
        }
    }

    return minCost;
}

// Example usage:
const nums = [3, 1, 4, 1, 5, 9, 2, 6, 5];
console.log(minCost(nums)); // Output should be 3
