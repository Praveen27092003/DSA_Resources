function getCircularElement(arr, index) {
    const n = arr.length;
    // Use modulo operator to wrap around the index
    return arr[((index % n) + n) % n];
}

// Example usage:
const arr = [1, 2, 3, 4, 5];

console.log(getCircularElement(arr, 0));  // Output: 1
console.log(getCircularElement(arr, 4));  // Output: 5
console.log(getCircularElement(arr, 5));  // Output: 1 (wrapped around)
console.log(getCircularElement(arr, -1)); // Output: 5 (wrapped around)
